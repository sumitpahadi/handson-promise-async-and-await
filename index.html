Question 3=>"Write the promise function to print numbers from 1 to 7, in which 1 should be printed after 1 sec, 2 should be printed after 2 sec, 3 should be printed after 3 sec, and so on. Numbers 1 2 3 4 5 6 7" 


      function printNumbers() {
        let number = 1;
      
        function printNumberWithDelay() {
          return new Promise(resolve => {
            setTimeout(() => {
              console.log(number);
              resolve();
              number++;
      
              if (number <= 7) {
                printNumberWithDelay();
              }
            }, number * 1000);
          });
        }
      
        printNumberWithDelay();
      }
      
      printNumbers();

Question 9=>Create examples to explain promise.all function


  const promise =new Promise((resolve ,reject)=>{
    setTimeout(()=>{
      resolve("promise 1")
    },1000)
  })
  const promise1 =new Promise((resolve ,reject)=>{
    setTimeout(()=>{
      resolve("promise 2")
    },2000)
  })
  const promise2 =new Promise((resolve ,reject)=>{
    setTimeout(()=>{
      resolve("promise 3")
    },4000)
  })
  const promise3 =new Promise((resolve ,reject)=>{
    setTimeout(()=>{
      resolve("promise 4")
    },4000)
  })
  
  Promise.all([promise,promise1,promise2,promise3])
  .then((result)=>{
    console.log("all promise fulllfilled,",result)
  })
